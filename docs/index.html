Key Prisma Models (AI & Chat) #

// ===== CHAT & CUSTOMER CARE =====
model Conversation {
  id            String   @id @default(cuid())
  userId        String?
  user          User?    @relation(fields: [userId], references: [id])
  title         String
  status        ConvStatus @default(OPEN)
  assignedTo    String?
  assignedStaff User?    @relation("StaffConversations", fields: [assignedTo], references: [id])
  messages      ChatMessage[]
  isFromChatbox Boolean @default(true)
  createdAt     DateTime @default(now())
  lastMessageAt DateTime @default(now())
  @@index([userId])
  @@index([status])
}

model ChatMessage {
  id             String     @id @default(cuid())
  conversationId String
  conversation   Conversation @relation(fields: [conversationId], references: [id])
  userId         String?
  user           User?       @relation(fields: [userId], references: [id])
  role           MsgRole
  content        String
  metadata       Json?
  isRead         Boolean   @default(false)
  createdAt      DateTime  @default(now())
  @@index([conversationId, createdAt])
}

// ===== USER BEHAVIOR & RECOMMENDATION =====
model UserBehavior {
  id        String     @id @default(cuid())
  userId    String
  user      User       @relation(fields: [userId], references: [id])
  action    BehaviorAction
  productId String?
  product   Product?   @relation(fields: [productId], references: [id])
  metadata  Json?
  sessionId String?
  createdAt DateTime  @default(now())
  @@index([userId, action, createdAt])
}

// ===== SIZE RECOMMENDATION =====
model SizeChart {
  id         String @id @default(cuid())
  categoryId String
  category   Category @relation(fields: [categoryId], references: [id])
  region     String
  gender     String?
  measurements Json
  sizeMap    Json?
  updatedAt  DateTime @updatedAt
  @@unique([categoryId, region])
}

// ===== PRODUCT EMBEDDINGS (pgvector) =====
// CREATE EXTENSION vector;
// CREATE INDEX idx_product_embedding ON "ProductEmbedding" USING ivfflat (embedding vector_cosine_ops);
model ProductEmbedding {
  id        String   @id @default(cuid())
  productId String   @unique
  product   Product  @relation(fields: [productId], references: [id])
  model     String   @default("text-embedding-3-small")
  updatedAt DateTime @updatedAt
}

Tại sao Zustand + TanStack Query? #

🔀

Phân chia rõ ràng

Query → Server state (products, orders, users)
Zustand → Client state (sidebar, filters, modal, chat UI)

📦

Bundle siêu nhẹ

Zustand ~3kB + TanStack Query ~13kB = 16kB
Nhẹ hơn RTK (12kB) + RTK Query (15kB) = 27kB

⚡

Performance

Zustand chỉ re-render component subscribe đúng state. Query auto cache, dedup, background refetch

🧩

DX

Zustand: create store 3 dòng. Query: useQuery tự động loading/error state. DevTools sẵn

Ví dụ code #

// TanStack Query: Server State
export const useOrders = (filters: OrderFilters) => {
  return useQuery({
    queryKey: ['orders', filters],
    queryFn: () => api.getOrders(filters),
  });
};

// Zustand: Client UI State
export const useUIStore = create<UIStore>((set) => ({
  sidebarOpen: true,
  selectedOrders: [],
  toggleSidebar: () => set((s) => ({ sidebarOpen: !s.sidebarOpen })),
}));

// Component
function OrderTable() {
  const filters = useUIStore((s) => s.orderFilters);
  const { data, isLoading } = useOrders(filters);
  if (isLoading) return <SkeletonTable />;
  return <DataTable rows={data?.orders} />;
}
